foreach Itera sobre los elementos del array la variable "deporte" va tomando el valor de esos elementos. 
                
                  let numeros = [2, 4, 6];
                  let listaAlCuadrado = [];

                  function llenarLista() {
                    numeros.forEach((numero) => {
                      return listaAlCuadrado.push(Math.pow(numero, 2));
                    });
                  }

                  llenarLista();

                  console.log(listaAlCuadrado);
map El método map() crea un nuevo array con los resultados de la llamada a la función indicada aplicados a cada uno de sus elementos. 
                
                  let numeros = [4, 9, 16, 25];
                  let raizCuadrada = numeros.map(Math.sqrt);

                  console.log(raizCuadrada);
concat El método concat() se usa para unir dos o más arrays. Este método no cambia los arrays existentes, sino que devuelve un nuevo array. 
                
                  let marvel = ["Black Widow", "Iron Man", "Captain America"];
                  let dc = ["Wonder Woman", "Batman", "Harley Quinn"];
                  let combinado = marvel.concat(dc);

                  console.log(combinado);
Spread Operator El operador de propagación spread operator permite que una expresión sea expandida en situaciones donde se esperan múltiples argumentos (llamadas a funciones) o múltiples elementos (arrays literales). 
                
                  let marvel = ["Black Widow", "Iron Man", "Captain America"];
                  let dc = ["Wonder Woman", "Batman", "Harley Quinn"];
                  let combinado = [...marvel, ...dc];

                  console.log(combinado);
También se utiliza para duplicar arreglos. 
                
                  let dc_duplicado = [...dc];

                  console.log(dc_duplicado);
Invertir un arreglo. 
                
                  let dc_reverse = [...dc].reverse();

                  console.log(dc_reverse);
Se puede utilizar para realizar operaciones. 
                
                  let multiplicacion = (num1, num2, num3) => {
                    console.log(num1 * num2 * num3);
                  };
                  
                  let numeros = [10, 10, 2];
                  multiplicacion(...numeros);
filter Itera sobre los elementos del arreglo y los filtra según la condición especificada. 
                
                  let numeros = [10, 14, 2, 100, 7, 25];
                  let mayorOIgualA10 = numeros.filter((numero) => numero >= 10);

                  console.log(mayorOIgualA10);
find El método find() devuelve el valor del primer elemento que cumple la función de prueba proporcionada. 
                
                  let campeonesUCL = [
                    {
                      nombre: "Real Madrid",
                      anio: 2018,
                      pais: "España",
                      cantidadDeVecesCampeon: 13,
                    },
                    {
                      nombre: "Liverpool",
                      anio: 2019,
                      pais: "Inglaterra",
                      cantidadDeVecesCampeon: 6,
                    },
                    {
                      nombre: "Bayern Munich",
                      anio: 2020,
                      pais: "Alemania",
                      cantidadDeVecesCampeon: 6,
                    },
                  ];

                  const campeon2019 = campeonesUCL.find((campeon) => {
                    return campeon.anio === 2019;
                  });

                  console.log(campeon2019.nombre);
reduce El método reduce() ejecuta una función reductora sobre cada elemento de un array, devolviendo como resultado un único valor. 
                
                  let cantidadDeVecesCampeones = campeonesUCL.reduce(
                    (cantidadTotal, campeon) => {
                      return cantidadTotal + campeon.cantidadDeVecesCampeon;
                  }, 0);

                  console.log(cantidadDeVecesCampeones);
